75void AllSpikingNeurons::advanceVertices(AllEdges &synapses, const EdgeIndexMap *edgeIndexMap) {
76 int maxSpikes = (int) ((Simulator::getInstance().getEpochDuration() * Simulator::getInstance().getMaxFiringRate()));
77
78 AllSpikingSynapses &spSynapses = dynamic_cast<AllSpikingSynapses &>(synapses);
79 // For each neuron in the network
80 for (int idx = Simulator::getInstance().getTotalVertices() - 1; idx >= 0; --idx) {
81 // advance neurons
82 advanceNeuron(idx);
83
84 // notify outgoing/incoming synapses if neuron has fired
85 if (hasFired_[idx]) {
86 LOG4CPLUS_DEBUG(vertexLogger_, "Neuron: " << idx << " has fired at time: "
87 << g_simulationStep * Simulator::getInstance().getDeltaT());
88
89 assert(spikeCount_[idx] < maxSpikes);
90
91 // notify outgoing synapses
92 BGSIZE synapseCounts;
93
94 if (edgeIndexMap != nullptr) {
95 synapseCounts = edgeIndexMap->outgoingEdgeCount_[idx];
96 if (synapseCounts != 0) {
97 int beginIndex = edgeIndexMap->outgoingEdgeBegin_[idx];
98 BGSIZE iEdg;
99 for (BGSIZE i = 0; i < synapseCounts; i++) {
100 iEdg = edgeIndexMap->outgoingEdgeIndexMap_[beginIndex + i];
101 spSynapses.preSpikeHit(iEdg);
102 }
103 }
104 }
105
106 // notify incoming synapses
107 synapseCounts = spSynapses.edgeCounts_[idx];
108 BGSIZE synapse_notified = 0;
109
110 if (spSynapses.allowBackPropagation()) {
111 for (int z = 0; synapse_notified < synapseCounts; z++) {
112 BGSIZE iEdg = Simulator::getInstance().getMaxEdgesPerVertex() * idx + z;
113 if (spSynapses.inUse_[iEdg] == true) {
114 spSynapses.postSpikeHit(iEdg);
115 synapse_notified++;
116 }
117 }
118 }
119
120 hasFired_[idx] = false;
121 }

128void AllSpikingNeurons::fire(const int index) const {
