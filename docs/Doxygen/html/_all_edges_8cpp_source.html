168 EdgeIndexMap *AllEdges::createEdgeIndexMap() {
169  int neuronCount = Simulator::getInstance().getTotalVertices();
170  int totalSynapseCount = 0;
171 
172  // count the total edges
173  for (int i = 0; i < neuronCount; i++) {
174  assert(static_cast<int>(synapseCounts_[i]) < Simulator::getInstance().getMaxEdgesPerVertex());
175  totalSynapseCount += synapseCounts_[i];
176  }
177 
178  DEBUG (cout << "totalSynapseCount: " << totalSynapseCount << endl;)
179 
180  if (totalSynapseCount == 0) {
181  return NULL;
182  }
183 
184  // allocate memories for forward map
185  vector<BGSIZE> *rgSynapseSynapseIndexMap = new vector<BGSIZE>[neuronCount];
186 
187  BGSIZE syn_i = 0;
188  int numInUse = 0;
189 
190  // create synapse forward map & active synapse map
191  EdgeIndexMap *edgeIndexMap = new EdgeIndexMap(neuronCount, totalSynapseCount);
192  for (int i = 0; i < neuronCount; i++) {
193  BGSIZE edge_count = 0;
194  edgeIndexMap->incomingSynapseBegin_[i] = numInUse;
195  for (int j = 0; j < Simulator::getInstance().getMaxEdgesPerVertex(); j++, syn_i++) {
196  if (inUse_[syn_i] == true) {
197  int idx = sourceNeuronIndex_[syn_i];
198  rgSynapseSynapseIndexMap[idx].push_back(syn_i);
199 
200  edgeIndexMap->incomingSynapseIndexMap_[numInUse] = syn_i;
201  numInUse++;
202  edge_count++;
203  }
204  }
205  assert(edge_count == this->synapseCounts_[i]);
206  edgeIndexMap->incomingSynapseCount_[i] = edge_count;
207  }
208 
209  assert(totalSynapseCount == numInUse);
210  this->totalEdgeCount_ = totalSynapseCount;
211 
212  syn_i = 0;
213  for (int i = 0; i < neuronCount; i++) {
214  edgeIndexMap->outgoingSynapseBegin_[i] = syn_i;
215  edgeIndexMap->outgoingSynapseCount_[i] = rgSynapseSynapseIndexMap[i].size();
216 
217  for (BGSIZE j = 0; j < rgSynapseSynapseIndexMap[i].size(); j++, syn_i++) {
218  edgeIndexMap->outgoingSynapseIndexMap_[syn_i] = rgSynapseSynapseIndexMap[i][j];
219  }
220  }
221 
222  // delete memories
223  delete[] rgSynapseSynapseIndexMap;
224 
225  return edgeIndexMap;
226 }
