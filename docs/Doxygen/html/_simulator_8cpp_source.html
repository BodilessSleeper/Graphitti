125 void Simulator::simulate() {
126  // Main simulation loop - execute maxGrowthSteps
127  for (int currentEpoch = 1; currentEpoch <= numEpochs_; currentEpoch++) {
128  LOG4CPLUS_TRACE(consoleLogger_, "Performing epoch number: " << currentEpoch);
129  LOG4CPLUS_TRACE(fileLogger_, "Begin network state:");
130  currentEpoch_ = currentEpoch;
131 #ifdef PERFORMANCE_METRICS
132  // Start timer for advance
133  short_timer.start();
134 #endif
135  // Advance simulation to next growth cycle
136  advanceUntilGrowth(currentEpoch);
137 #ifdef PERFORMANCE_METRICS
138  // Time to advance
139  t_host_advance += short_timer.lap() / 1000000.0;
140 #endif
141  LOG4CPLUS_TRACE(consoleLogger_, "done with epoch cycle " << currentEpoch_ << ", beginning growth update");
142 
143 #ifdef PERFORMANCE_METRICS
144  // Start timer for connection update
145  short_timer.start();
146 #endif
147  // Update the neuron network
148  model_->updateConnections();
149  model_->updateHistory();
150 
151 #ifdef PERFORMANCE_METRICS
152  // Times converted from microseconds to seconds
153  // Time to update synapses
154  t_host_adjustSynapses += short_timer.lap() / 1000000.0;
155  // Time since start of simulation
156  double total_time = timer.lap() / 1000000.0;
157 
158  cout << "\ntotal_time: " << total_time << " seconds" << endl;
159  printPerformanceMetrics(total_time, currentEpoch);
160  cout << endl;
161 #endif
162  }

168 void Simulator::advanceUntilGrowth(const int &currentEpoch) const {
169  uint64_t count = 0;
170  // Compute step number at end of this simulation epoch
171  uint64_t endStep = g_simulationStep
172  + static_cast<uint64_t>(epochDuration_ / deltaT_);
173  // DEBUG_MID(model->logSimStep();) // Generic model debug call
174  while (g_simulationStep < endStep) {
175  // Output status once every 10,000 steps
176  if (count % 10000 == 0) {
177  LOG4CPLUS_TRACE(consoleLogger_, "Epoch: " << currentEpoch << "/" << numEpochs_
178  << " simulating time: "
179  << g_simulationStep * deltaT_ << "/"
180  << (epochDuration_ * numEpochs_) - 1);
181  count = 0;
182  }
183  count++;

188  // Advance the Network one time step
