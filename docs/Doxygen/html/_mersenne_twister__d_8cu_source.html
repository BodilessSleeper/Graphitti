105 // Write MT_RNG_COUNT vertical lanes of nPerRng random numbers to *d_Random.
106 // For coalesced global writes MT_RNG_COUNT should be a multiple of warp size.
107 // Initial states for each generator are the same, since the states are
108 // initialized from the global seed. In order to improve distribution properties
109 // on small NPerRng supply dedicated (local) seed to each twister.
110 // The local seeds, in their turn, can be extracted from global seed
111 // by means of any simple random number generator, like LCG.
113 __global__ void RandomGPU(
114  float *d_Random,
115  int nPerRng, int mt_rng_count)
116 {
117  const int tid = blockDim.x * blockIdx.x + threadIdx.x;
118  int iState, iState1, iStateM, iOut;
119  unsigned int mti, mti1, mtiM, x;
120  unsigned int matrix_a, mask_b, mask_c;
121 
122  //Load bit-vector Mersenne Twister parameters
123  matrix_a = ds_MT[tid].matrix_a;
124  mask_b = ds_MT[tid].mask_b;
125  mask_c = ds_MT[tid].mask_c;
126 
127  iState = ds_MT[tid].iState;
128  mti1 = mt[iState];
129  for (iOut = 0; iOut < nPerRng; iOut++) {
130  iState1 = iState + 1;
131  iStateM = iState + MT_MM;
132  if(iState1 >= MT_NN*(1+tid)) iState1 -= MT_NN;
133  if(iStateM >= MT_NN*(1+tid)) iStateM -= MT_NN;
134  mti = mti1;
135  mti1 = mt[iState1];
136  mtiM = mt[iStateM];
137 
138  // MT recurrence
139  x = (mti & MT_UMASK) | (mti1 & MT_LMASK);
140  x = mtiM ^ (x >> 1) ^ ((x & 1) ? matrix_a : 0);
141 
142  mt[iState] = x;
143  iState = iState1;
144 
145  //Tempering transformation
146  x ^= (x >> MT_SHIFT0);
147  x ^= (x << MT_SHIFTB) & mask_b;
148  x ^= (x << MT_SHIFTC) & mask_c;
149  x ^= (x >> MT_SHIFT1);
150 
151  //Convert to (0, 1] float and write to global memory
152  d_Random[tid + iOut * mt_rng_count] = ((float)x + 1.0f) / 4294967296.0f;
153  }
154  ds_MT[tid].iState = iState;
155 }

158 // Transform each of MT_RNG_COUNT lanes of nPerRng uniformly distributed
159 // random samples, produced by RandomGPU(), to normally distributed lanes
160 // using Cartesian form of Box-Muller transformation.
161 // nPerRng must be even.
163 #define PI 3.14159265358979f
164 __device__ inline void BoxMuller(float& u1, float& u2){
165  float r = sqrtf(-2.0f * logf(u1));
166  float phi = 2 * PI * u2;
167  u1 = r * __cosf(phi);
168  u2 = r * __sinf(phi);
169 }

171 __global__ void BoxMullerGPU(float *d_Random, int nPerRng, int mt_rng_count){
172  const int tid = blockDim.x * blockIdx.x + threadIdx.x;
173 
174  for (int iOut = 0; iOut < nPerRng; iOut += 2)
175  BoxMuller(
176  d_Random[tid + (iOut + 0) * mt_rng_count],
177  d_Random[tid + (iOut + 1) * mt_rng_count]
178  );
179 }

184 __global__ void RandomNormGPU(
185  float *d_Random,
186  int nPerRng, int mt_rng_count)
187 {
188  const int tid = blockDim.x * blockIdx.x + threadIdx.x;
189  int iState, iState1, iStateM, iOut;
190  unsigned int mti, mti1, mtiM, x;
191  unsigned int matrix_a, mask_b, mask_c;
192 
193  float regVal1, regVal2; //need 2 values for boxmuller
194  bool boxFlag = false; //will perform boxmuller transform on true
195 
196  //Load bit-vector Mersenne Twister parameters
197  matrix_a = ds_MT[tid].matrix_a;
198  mask_b = ds_MT[tid].mask_b;
199  mask_c = ds_MT[tid].mask_c;
200 
201  iState = ds_MT[tid].iState;
202  mti1 = mt[iState];
203  for (iOut = 0; iOut < nPerRng; iOut++) {
204  iState1 = iState + 1;
205  iStateM = iState + MT_MM;
206  if(iState1 >= MT_NN*(1+tid)) iState1 -= MT_NN;
207  if(iStateM >= MT_NN*(1+tid)) iStateM -= MT_NN;
208  mti = mti1;
209  mti1 = mt[iState1];
210  mtiM = mt[iStateM];
211 
212  // MT recurrence
213  x = (mti & MT_UMASK) | (mti1 & MT_LMASK);
214  x = mtiM ^ (x >> 1) ^ ((x & 1) ? matrix_a : 0);
215 
216  mt[iState] = x;
217  iState = iState1;
218 
219  //Tempering transformation
220  x ^= (x >> MT_SHIFT0);
221  x ^= (x << MT_SHIFTB) & mask_b;
222  x ^= (x << MT_SHIFTC) & mask_c;
223  x ^= (x >> MT_SHIFT1);
224 
225  if(boxFlag){
226  regVal2 = ((float)x + 1.0f) / 4294967296.0f;
227  BoxMuller(regVal1,regVal2);
228  d_Random[tid + (iOut-1) * mt_rng_count] = regVal1;
229  d_Random[tid + iOut * mt_rng_count] = regVal2;
230  boxFlag = false;
231  }else{
232  regVal1 = ((float)x + 1.0f) / 4294967296.0f;
233  boxFlag = true;
234  }
235  }
236  ds_MT[tid].iState = iState;
237 }
