399 void AllSTDPSynapses::stdpLearning(const BGSIZE iEdg, double delta, double epost, double epre) {
400  BGFLOAT STDPgap_ = this->STDPgap_[iEdg];
401  BGFLOAT muneg_ = this->muneg_[iEdg];
402  BGFLOAT mupos_ = this->mupos_[iEdg];
403  BGFLOAT tauneg_ = this->tauneg_[iEdg];
404  BGFLOAT taupos_ = this->taupos_[iEdg];
405  BGFLOAT Aneg_ = this->Aneg_[iEdg];
406  BGFLOAT Apos_ = this->Apos_[iEdg];
407  BGFLOAT Wex_ = this->Wex_[iEdg];
408  BGFLOAT &W = this->W_[iEdg];
409  synapseType type = this->type_[iEdg];
410  BGFLOAT dw;
411 
412  if (delta < -STDPgap_) {
413  // depression
414  dw = pow(fabs(W) / Wex_, muneg_) * Aneg_ * exp(delta / tauneg_); // normalize
415  } else if (delta > STDPgap_) {
416  // potentiation
417  dw = pow(fabs(Wex_ - fabs(W)) / Wex_, mupos_) * Apos_ * exp(-delta / taupos_); // normalize
418  } else {
419  return;
420  }
421 
422  // dw is the percentage change in synaptic strength; add 1.0 to become the scaling ratio
423  dw = 1.0 + dw * epre * epost;
424 
425  // if scaling ratio is less than zero, set it to zero so this synapse, its strength is always zero
426  if (dw < 0) {
427  dw = 0;
428  }
429 
430  // current weight multiplies dw (scaling ratio) to generate new weight
431  W *= dw;
432 
433  // if new weight is bigger than Wex_ (maximum allowed weight), then set it to Wex_
434  if (fabs(W) > Wex_) {
435  W = edgSign(type) * Wex_;
436  }
437 
438  LOG4CPLUS_DEBUG(fileLogger_,
439  "AllSTDPSynapses::stdpLearning:" << endl
440  << "\tiSyn: " << iEdg << endl
441  << "\tdelta: " << delta << endl
442  << "\tepre: " << epre << endl
443  << "\tepost: " << epost << endl
444  << "\tdw: " << dw << endl
445  << "\tW: " << W << endl << endl);
446 }

452 bool AllSTDPSynapses::isSpikeQueuePost(const BGSIZE iEdg) {
