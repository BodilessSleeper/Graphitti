95void AllEdges::setupEdges(const int numVertices, const int maxEdges) {
96 BGSIZE maxTotalEdges = maxEdges * numVertices;
97
98 maxEdgesPerVertex_ = maxEdges;
99 totalEdgeCount_ = 0;
100 countVertices_ = numVertices;
101
102 if (maxTotalEdges != 0) {
103 destVertexIndex_ = new int[maxTotalEdges];
104 W_ = new BGFLOAT[maxTotalEdges];
105 summationPoint_ = new BGFLOAT *[maxTotalEdges];
106 sourceVertexIndex_ = new int[maxTotalEdges];
107 type_ = new edgeType[maxTotalEdges];
108 inUse_ = new bool[maxTotalEdges];
109 edgeCounts_ = new BGSIZE[numVertices];
110
111 for (BGSIZE i = 0; i < maxTotalEdges; i++) {
112 summationPoint_[i] = nullptr;
113 inUse_[i] = false;
114 W_[i] = 0;
115 }
116
117 for (int i = 0; i < numVertices; i++) {
118 edgeCounts_[i] = 0;
119 }
120 }

185void AllEdges::createEdgeIndexMap(shared_ptr<EdgeIndexMap> edgeIndexMap) {
186 Simulator& simulator = Simulator::getInstance();
187 int vertexCount = simulator.getTotalVertices();
188 int totalEdgeCount = 0;
189
190 // count the total edges
191 for (int i = 0; i < vertexCount; i++) {
192 assert(static_cast<int>(edgeCounts_[i]) < simulator.getMaxEdgesPerVertex());
193 totalEdgeCount += edgeCounts_[i];
194 }
195
196 LOG4CPLUS_TRACE(fileLogger_,endl<<"totalEdgeCount: in edgeIndexMap " << totalEdgeCount << endl);
197
198 // Create vector for edge forwarding map
199 vector<BGSIZE> rgEdgeEdgeIndexMap[vertexCount];
200
201 BGSIZE edg_i = 0;
202 int curr = 0;
203
204 LOG4CPLUS_TRACE(edgeLogger_, "\nSize of edge Index Map "<< vertexCount << "," << totalEdgeCount << endl);
205
206 for (int i = 0; i < vertexCount; i++) {
207 BGSIZE edge_count = 0;
208 edgeIndexMap->incomingEdgeBegin_[i] = curr;
209 for (int j = 0; j < simulator.getMaxEdgesPerVertex(); j++, edg_i++) {
210 if (inUse_[edg_i]) {
211 int idx = sourceVertexIndex_[edg_i];
212 rgEdgeEdgeIndexMap[idx].push_back(edg_i);
213
214 edgeIndexMap->incomingEdgeIndexMap_[curr] = edg_i;
215 curr++;
216 edge_count++;
217 }
218 }
219
220 if(edge_count != edgeCounts_[i])
221 {
222 LOG4CPLUS_FATAL(edgeLogger_, "\nedge_count does not match edgeCounts_" << edge_count << endl);
223 throw runtime_error("createEdgeIndexMap: edge_count does not match edgeCounts_.");
224 }
225
226 edgeIndexMap->incomingEdgeCount_[i] = edge_count;
227 }
228
229 if(totalEdgeCount != curr)
230 {
231 LOG4CPLUS_FATAL(edgeLogger_,"Curr does not match the totalEdgeCount. curr are " << curr << endl);
232 throw runtime_error("createEdgeIndexMap: Curr does not match the totalEdgeCount.");
233 }
234 totalEdgeCount_ = totalEdgeCount;
235 LOG4CPLUS_DEBUG(edgeLogger_,endl<<"totalEdgeCount: " << totalEdgeCount_ << endl);
236
237 edg_i = 0;
238 for (int i = 0; i < vertexCount; i++) {
239 edgeIndexMap->outgoingEdgeBegin_[i] = edg_i;
240 edgeIndexMap->outgoingEdgeCount_[i] = rgEdgeEdgeIndexMap[i].size();
241
242 for (BGSIZE j = 0; j < rgEdgeEdgeIndexMap[i].size(); j++, edg_i++) {
243 edgeIndexMap->outgoingEdgeIndexMap_[edg_i] = rgEdgeEdgeIndexMap[i][j];
244 }
245 }
246}

284void AllEdges::addEdge(BGSIZE &iEdg, edgeType type, const int srcVertex, const int destVertex, BGFLOAT *sumPoint,
285 const BGFLOAT deltaT) {
286 if (edgeCounts_[destVertex] >= maxEdgesPerVertex_) {
287 LOG4CPLUS_FATAL(edgeLogger_, "Vertex : " << destVertex << " ran out of space for new edges.");
288 throw runtime_error("Vertex : " + destVertex + string(" ran out of space for new edges."));
289 }
290
291 // add it to the list: find first edge location for vertex destVertex
292 // that isn't in use.
293 for (BGSIZE i = 0; i < maxEdgesPerVertex_; i++) {
294 iEdg = maxEdgesPerVertex_ * destVertex + i;
295 if (!inUse_[iEdg]) {
296 break;
297 }
298 }
299
300 edgeCounts_[destVertex]++;
301
302 // create an edge
303 createEdge(iEdg, srcVertex, destVertex, sumPoint, deltaT, type);
304}
