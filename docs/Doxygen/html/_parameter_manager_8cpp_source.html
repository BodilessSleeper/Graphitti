101bool ParameterManager::getStringByXpath(string xpath, string &referenceVar) {
102 if (!checkDocumentStatus()) return false;

110 return false;

123bool ParameterManager::getIntByXpath(string xpath, int &referenceVar) {
124 if (!checkDocumentStatus()) return false;
125 string tmp;
126 if (!getStringByXpath(xpath, tmp)) {
127 return false;
128 }
129 // Workaround for standard value conversion functions.
130 // stoi() will cast floats to ints.
131 if (regex_match(tmp, regex("\\d+[.]\\d+(e[+-]?\\d+)?f?|\\d+[.]?\\d+(e[+-]?\\d+)?f"))) {
132 cerr << "Parsed parameter is likely a float/double value. "
133 << "Terminating integer cast. Value: "
134 << tmp << endl;
135 return false;
136 } else if (regex_match(tmp, regex(".*[^\\def.]+.*"))) {
137 cerr << "Parsed parameter is likely a string. "
138 << "Terminating integer cast. Value: "
139 << tmp << endl;
140 return false;
141 }
142 try {
143 referenceVar = stoi(tmp);
144 } catch (invalid_argument &arg_exception) {
145 cerr << "Parsed parameter could not be parsed as an integer. Value: "
146 << tmp << endl;
147 return false;
148 } catch (out_of_range &range_exception) {
149 cerr << "Parsed string parameter could not be converted to an integer. Value: "
150 << tmp << endl;
151 return false;
152 }
153 return true;
154}

291bool ParameterManager::getIntVectorByXpath(const string &path, const string &elementName, vector<int> &referenceVar) {
292 // Open file using a local XmlDocument object
293 TiXmlDocument xmlDocument(path.c_str());
294 if (!xmlDocument.LoadFile()) {
295 cerr << "Failed to load " << path.c_str() << ":" << "\n\t"
296 << xmlDocument.ErrorDesc() << endl;
297 return false;
298 }
299
300 // Check file for matching element
301 TiXmlNode *xmlNode = nullptr;
302 if ((xmlNode = xmlDocument.FirstChildElement(elementName)) == nullptr) {
303 cerr << "Could not find <" << elementName << "> in vertex list file " << path << endl;
304 return false;
305 }
306
307 // Get list of ints as a string stream
308 std::istringstream valueStream(xmlNode->ToElement()->GetText());
309
310 // Parse integers out of the string and add them to a list
311 int i;
312 while (valueStream.good()) {
313 valueStream >> i;
314 referenceVar.push_back(i);
315 }
316 return true;
317}
