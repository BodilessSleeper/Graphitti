181void AllDynamicSTDPSynapses::changePSR(const BGSIZE iEdg, const BGFLOAT deltaT) {
182 BGFLOAT &psr = this->psr_[iEdg];
183 BGFLOAT &W = this->W_[iEdg];
184 BGFLOAT &decay = this->decay_[iEdg];
185 uint64_t &lastSpike = this->lastSpike_[iEdg];
186 BGFLOAT &r = this->r_[iEdg];
187 BGFLOAT &u = this->u_[iEdg];
188 BGFLOAT &D = this->D_[iEdg];
189 BGFLOAT &F = this->F_[iEdg];
190 BGFLOAT &U = this->U_[iEdg];
191
192 // adjust synapse parameters
193 if (lastSpike != ULONG_MAX) {
194 BGFLOAT isi = (g_simulationStep - lastSpike) * deltaT;
195 r = 1 + (r * (1 - u) - 1) * exp(-isi / D);
196 u = U + u * (1 - U) * exp(-isi / F);
197 }
198 psr += ((W / decay) * u * r); // calculate psr
199 lastSpike = g_simulationStep; // record the time of the spike
200}
